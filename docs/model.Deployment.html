class Deployment(Base):

View In Hierarchy

Represents a Deployment.

Example

To create a deployment:

container_spec_1 = {
"NAME": "WebServer", "LENGTH": (1000, 1500), "CPU": (1.0, 2.0), "RAM": (512, 1024), "GPU": None

}

container_spec_2 = {
"NAME": "Database", "LENGTH": (2000, 2500), "CPU": (0.5, 1.5), "RAM": (256, 512), "GPU": (1, 2)

}

container_specs = [container_spec_1, container_spec_2]

deployment = Deployment(NAME="MyDeployment", replicas=3, CONTAINER_SPECS=container_specs)
